Regression as distributions

\[ y_i \sim \mathcal{N}{(\mu_i, \sigma)} \]

\[ \mu_i = \beta_0+\beta_1*X_i \]

\[ \beta_0 = 5 \]

\[ \beta_1 = 2 \]

n_obs = 100
x = rep(seq(0,10,1),n_obs)
intercept = 5
slope = 2
error = rnorm(n_obs,0,5)
y = intercept+slope*x+error
m1 = lm(y~x)
tab_model(m1, show.ci = F, show.se = T)
